Callbacks, Closures

Callback

Callback - функция обратного вызова - ф-ция, которая передаётся другой ф-ции как аргумент

const doMath = function (a, b, callback) {
const result = callback(a, b);
console.log(result);
}
const add = function (x, y) {
return x + y;
}
doMath(2, 3, add);

Closure

Closure - замыкание - Возвращаемая функция во время вызова будет иметь доступ ко всем локальным переменным (области видимости) родительской функции (той, из которой её вернули)

const makeSheff = function (name) {
const innerVar = "cooks";

const makeDish = function (dish) {
console.log(`${name} ${innerVar} ${dish}`);
};

return makeDish;
};

const mango = makeSheff("Mango"); // создаём повара Mango

mango("tea"); // Mango cooks tea
mango("coffee"); // Mango cooks coffee
